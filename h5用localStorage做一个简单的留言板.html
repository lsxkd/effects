html代码如下：

	<div>
		<textarea name="" id="trCon" cols="30" rows="10"></textarea>
	</div>
	<div>
		<input type="button" value="发表" onclick="PostCon()">
		<input type="button" value="清除" onclick="ClearCon()">
	</div>
	<div id="dCon"></div>

js代码如下：

	window.onload=function(){
		if(localStorage.getItem("k_con")!=null){
			document.getElementById("dCon").innerHTML=localStorage.getItem("k_con")
		}
	}
	function PostCon(){
		var htmls=document.getElementById("dCon").innerHTML;
		var trCon=document.getElementById("trCon").value;
		var dCon=document.getElementById("dCon");
		if(trCon == ""){
			alert("请输入内容")
			return false
		}
		htmls+="<p>" + trCon + "</p>"
		dCon.innerHTML = htmls;
		localStorage.setItem("k_con",dCon.innerHTML)
	}
	function ClearCon(){
		document.getElementById("trCon").value = ""
		document.getElementById("dCon").innerHTML=""
		localStorage.clear();
	}
